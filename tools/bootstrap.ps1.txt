from pathlib import Path
r = Path('.').resolve()
for p in r.rglob('*'):
    if '.venv' in p.parts or '__pycache__' in p.parts: continue
    if p.suffix.lower() not in {'.py', '.ps1', '.csv', '.bat', '.mdc', '.json'}: continue
    d = p.read_bytes()
    if b'\x00' not in d and not d.startswith(b'\xff\xfe'): continue
    enc = 'utf-16' if d.startswith(b'\xff\xfe') else 'utf-16-le'
    p.write_text(d.decode(enc), encoding='utf-8', newline='\n')
    print('fixed', p)
"@

function Write-Utf8Ps1($relPath, $content) {
    $path = Join-Path (Get-Location) $relPath
    $dir = Split-Path $path -Parent
    if ($dir -and -not (Test-Path $dir)) { New-Item -ItemType Directory -Path $dir | Out-Null }
    $utf8bom = New-Object System.Text.UTF8Encoding $true
    [System.IO.File]::WriteAllText($path, ($content.TrimEnd() + "`r`n"), $utf8bom)
    Write-Host "wrote $path"
}
